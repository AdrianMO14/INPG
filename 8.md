#include <stdio.h>
#define ROZ 50
#define ZNAK 'a'


char * isElement(char*tab,char znak)
{
    int i=0;
    while(tab[i]!='\0')
    {
        if(tab[i]==znak)
            return &tab[i];
        else
            i++;
    }
    return NULL;
}

int main()
{

    char tab[ROZ];
    char *p;
    fgets(tab,ROZ,stdin);

    p=isElement(tab,ZNAK);

    if(p==NULL)
        printf("Nie ma tego znaku");
    else
        printf("%c\n",*p);

    return 0;
}
