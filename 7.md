#include <stdio.h>
#include <stdlib.h>
#define ROW1 2
#define COL1 3
#define ROW2 3
#define COL2 2
int main()
{
    int mac1[ROW1][COL1]={
    {1,0,2},
    {-1,3,1},
    };

    int mac2[ROW2][COL2]={
    {3,1},
    {2,1},
    {1,0}

    };
    int mac3[ROW1][COL2]={
    {0,0},
    {0,0}

    };

    int wiersz,kolumna,i;

    for(wiersz=0;wiersz<ROW1;wiersz++)
        {
            for(kolumna=0;kolumna<COL2;kolumna++)
                {
                    for(i=0;i<ROW2;i++)
                        {
                            mac3[wiersz][kolumna]=mac1[wiersz][i]*mac2[i][kolumna]+mac3[wiersz][kolumna];
                        }
                }


        }
    int w,k;
    for(w=0;w<ROW1;w++)
        {
        for(k=0;k<COL2;k++)
            {
                printf(" %i ",mac3[w][k]);
            }
         printf("\n");
        }

    return 0;
}
