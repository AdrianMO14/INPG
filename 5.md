#include <stdlib.h>
#include <stdio.h>
#define ZAKRES 20
#define ROZMIAR 10      //zminaiadasdasdas
int main()
{

    float tab[ROZMIAR];
    float pomoc;
    void srand();
    int rand();
    srand(20);
    int i;
    while(i!=ROZMIAR)
        {
            tab[i]=rand()%ZAKRES+1;
            i++;
        }

    int m=0;
    int j=0;
    while(m!=ROZMIAR)
        {

            while(j!=ROZMIAR)
                {
                    if(tab[m]>tab[j])
                        {
                            pomoc=tab[m];
                            tab[m]=tab[j];
                            tab[j]=pomoc;
                            j++;
                        }
                    else
                        j++;

                }
                j=m+1;
                m++;
        }

    int k=0;
    while(k!=ROZMIAR)
        {
            printf("%f\n",tab[k]);
            k++;
        }
    return 0;
}
